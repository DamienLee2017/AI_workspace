// 生成 G 代码核心算法
function generateGCode(params: MachiningParams): string {
  const lines: string[] = [];
  const {
    feedRate,
    spindleSpeed,
    cutDepth,
    stepOver,
    toolDiameter,
    safeHeight,
    modelBounds,
  } = params;

  // 文件头
  lines.push("; Generated by STEP to G-Code Converter");
  lines.push(`; Generated at: ${new Date().toISOString()}`);
  lines.push("; Feed Rate: " + feedRate + " mm/min");
  lines.push("; Spindle Speed: " + spindleSpeed + " rpm");
  lines.push("; Cut Depth: " + cutDepth + " mm");
  lines.push("; Step Over: " + stepOver + " mm");
  lines.push("; Tool Diameter: " + toolDiameter + " mm");
  lines.push("");

  // 初始化
  lines.push("G90 ; 绝对坐标编程");
  lines.push("G21 ; 公制单位 (mm)");
  lines.push("G17 ; XY 平面选择");
  lines.push("G40 ; 取消刀具半径补偿");
  lines.push("G49 ; 取消刀具长度补偿");
  lines.push("G80 ; 取消固定循环");
  lines.push("");

  // 主轴启动
  lines.push(`M3 S${spindleSpeed} ; 主轴正转，转速 ${spindleSpeed} rpm`);
  lines.push(`G4 P3 ; 等待主轴达到目标转速`);
  lines.push("");

  // 计算层数
  const modelHeight = modelBounds.maxZ - modelBounds.minZ;
  const numLayers = Math.ceil(modelHeight / cutDepth);
  
  // 刀具补偿
  lines.push(`G41 D1 ; 刀具半径左补偿，刀具直径 ${toolDiameter}`);
  lines.push("");

  // Z轴下刀
  lines.push(`G0 Z${safeHeight.toFixed(3)} ; 快速移动到安全高度`);
  lines.push(`X${modelBounds.minX.toFixed(3)} Y${modelBounds.minY.toFixed(3)} ; 移动到起点`);
  lines.push("");

  // 分层切削
  for (let layer = 0; layer < numLayers; layer++) {
    const currentZ = modelBounds.minZ + layer * cutDepth;
    const nextZ = Math.min(currentZ + cutDepth, modelBounds.maxZ);
    
    lines.push(`; === 第 ${layer + 1}/${numLayers} 层，Z = ${currentZ.toFixed(3)} mm ===`);
    
    // 下刀
    lines.push(`G1 Z${currentZ.toFixed(3)} F${(feedRate / 2).toFixed(0)} ; 慢速下刀`);
    
    // 螺旋下刀或斜线下刀
    if (layer === 0) {
      // 第一层使用螺旋下刀
      const centerX = (modelBounds.minX + modelBounds.maxX) / 2;
      const centerY = (modelBounds.minY + modelBounds.maxY) / 2;
      const helixRadius = Math.min(
        (modelBounds.maxX - modelBounds.minX) / 4,
        (modelBounds.maxY - modelBounds.minY) / 4
      );
      
      // 螺旋切削一圈
      lines.push(`G2 I${helixRadius.toFixed(3)} J0 Z${nextZ.toFixed(3)} F${feedRate} ; 螺旋下刀`);
    }
    
    // 平面内切削 - 沿着边界进行轮廓切削
    const contour = generateContourPath(modelBounds, stepOver);
    for (const point of contour) {
      lines.push(`G1 X${point.x.toFixed(3)} Y${point.y.toFixed(3)} Z${currentZ.toFixed(3)} F${feedRate}`);
    }
    
    // 抬刀
    lines.push(`G0 Z${safeHeight.toFixed(3)} ; 抬刀到安全高度`);
    lines.push("");
  }

  // 返回原点
  lines.push("G0 Z50 ; 抬刀到换刀位置");
  lines.push("G0 X0 Y0 ; 返回工件原点");
  lines.push("");

  // 主轴停止
  lines.push("M5 ; 主轴停止");
  lines.push("M30 ; 程序结束");
  lines.push("%");

  return lines.join("\n");
}

// 生成轮廓路径
function generateContourPath(
  bounds: { minX: number; minY: number; minZ: number; maxX: number; maxY: number; maxZ: number },
  stepOver: number
): { x: number; y: number }[] {
  const path: { x: number; y: number }[] = [];
  
  const width = bounds.maxX - bounds.minX;
  const height = bounds.maxY - bounds.minY;
  
  // 计算偏移量（考虑刀具半径）
  const offset = stepOver;
  
  // 外轮廓矩形
  const x1 = bounds.minX + offset;
  const y1 = bounds.minY + offset;
  const x2 = bounds.maxX - offset;
  const y2 = bounds.maxY - offset;
  
  // 简单的矩形轮廓
  path.push({ x: x1, y: y1 });
  path.push({ x: x2, y: y1 });
  path.push({ x: x2, y: y2 });
  path.push({ x: x1, y: y2 });
  path.push({ x: x1, y: y1 }); // 闭合轮廓
  
  return path;
}

// 计算 G 代码统计信息
function calculateGCodeStats(gcode: string): {
  totalLines: number;
  estimatedTime: number;
  toolpathLength: number;
} {
  const lines = gcode.split("\n").filter((line) => line.trim() && !line.startsWith(";"));
  
  let moveCount = 0;
  let totalDistance = 0;
  
  let currentX = 0,
    currentY = 0,
    currentZ = 0;
  
  for (const line of lines) {
    const gMatch = line.match(/G[01]\s+X([-\d.]+)\s+Y([-\d.]+)/i);
    const g0Match = line.match(/G0\s+X([-\d.]+)\s+Y([-\d.]+)/i);
    const g1Match = line.match(/G1\s+X([-\d.]+)\s+Y([-\d.]+)/i);
    
    let newX = currentX,
      newY = currentY;
    
    if (g0Match || g1Match) {
      newX = parseFloat(gMatch?.[1] || g0Match?.[1] || g1Match?.[1] || currentX.toString());
      newY = parseFloat(gMatch?.[2] || g0Match?.[2] || g1Match?.[2] || currentY.toString());
      
      const dx = newX - currentX;
      const dy = newY - currentY;
      totalDistance += Math.sqrt(dx * dx + dy * dy);
      
      currentX = newX;
      currentY = newY;
      moveCount++;
    }
    
    // Z轴移动
    const zMatch = line.match(/Z([-\d.]+)/i);
    if (zMatch) {
      const newZ = parseFloat(zMatch[1]);
      totalDistance += Math.abs(newZ - currentZ);
      currentZ = newZ;
    }
  }
  
  // 估算加工时间（秒）
  const estimatedTime = (totalDistance / 800) * 60; // 假设平均进给速度 800 mm/min
  
  return {
    totalLines: lines.length,
    estimatedTime: Math.round(estimatedTime),
    toolpathLength: Math.round(totalDistance),
  };
}

"use client";

import { useRef, useMemo, useEffect, useState } from "react";
import { Canvas, useFrame, useThree } from "@react-three/fiber";
import { OrbitControls, Grid, Axes3D, Environment } from "@react-three/drei";
import * as THREE from "three";

interface ToolpathPoint {
  x: number;
  y: number;
  z: number;
  type?: "rapid" | "feed" | "plunge" | "retract";
}

interface ToolpathPreviewProps {
  toolpath?: ToolpathPoint[];
  modelBounds?: {
    minX: number;
    maxX: number;
    minY: number;
    maxY: number;
    minZ: number;
    maxZ: number;
  };
  showGrid?: boolean;
  showAxes?: boolean;
}

function ToolpathLine({ toolpath }: { toolpath: ToolpathPoint[] }) {
  const lineRef = useRef<THREE.LineSegments>(null);

  const geometry = useMemo(() => {
    if (toolpath.length < 2) return null;

    const positions: number[] = [];
    const colors: number[] = [];

    const colorMap: Record<string, [number, number, number]> = {
      rapid: [1, 1, 0],    // 黄色 - 快进
      feed: [0, 1, 0],     // 绿色 - 切削
      plunge: [1, 0, 0],   // 红色 - 下刀
      retract: [0, 0.5, 1], // 蓝色 - 退刀
    };

    for (let i = 0; i < toolpath.length - 1; i++) {
      const p1 = toolpath[i];
      const p2 = toolpath[i + 1];

      positions.push(p1.x, p1.z, p1.y);
      positions.push(p2.x, p2.z, p2.y);

      const color = colorMap[p1.type || "feed"];
      colors.push(...color, ...color);
    }

    const geo = new THREE.BufferGeometry();
    geo.setAttribute("position", new THREE.Float32BufferAttribute(positions, 3));
    geo.setAttribute("color", new THREE.Float32BufferAttribute(colors, 3));

    return geo;
  }, [toolpath]);

  return geometry ? (
    <lineSegments ref={lineRef} geometry={geometry}>
      <lineBasicMaterial vertexColors transparent opacity={0.9} />
    </lineSegments>
  ) : null;
}

function ToolpathPoints({ toolpath }: { toolpath: ToolpathPoint[] }) {
  const points = useMemo(() => {
    if (toolpath.length === 0) return null;

    const positions = new Float32Array(toolpath.length * 3);
    const colors = new Float32Array(toolpath.length * 3);

    const colorMap: Record<string, [number, number, number]> = {
      rapid: [1, 1, 0],
      feed: [0, 1, 0],
      plunge: [1, 0, 0],
      retract: [0, 0.5, 1],
    };

    toolpath.forEach((p, i) => {
      positions[i * 3] = p.x;
      positions[i * 3 + 1] = p.z;
      positions[i * 3 + 2] = p.y;
      const color = colorMap[p.type || "feed"];
      colors[i * 3] = color[0];
      colors[i * 3 + 1] = color[1];
      colors[i * 3 + 2] = color[2];
    });

    const geometry = new THREE.BufferGeometry();
    geometry.setAttribute("position", new THREE.Float32BufferAttribute(positions, 3));
    geometry.setAttribute("color", new THREE.Float32BufferAttribute(colors, 3));

    return geometry;
  }, [toolpath]);

  return points ? (
    <points geometry={points}>
      <pointsMaterial vertexColors size={2} sizeAttenuation />
    </points>
  ) : null;
}

function WorkpieceModel({ bounds }: { bounds: ToolpathPreviewProps["modelBounds"] }) {
  const meshRef = useRef<THREE.Mesh>(null);

  const geometry = useMemo(() => {
    if (!bounds) return null;

    const { minX, maxX, minY, maxY, minZ, maxZ } = bounds;
    const width = maxX - minX;
    const height = maxY - minY;
    const depth = maxZ - minZ;

    const geo = new THREE.BoxGeometry(width, height, depth);
    geo.translate(
      (minX + maxX) / 2,
      (minY + maxY) / 2,
      (minZ + maxZ) / 2
    );

    return geo;
  }, [bounds]);

  return geometry ? (
    <mesh ref={meshRef} geometry={geometry}>
      <meshStandardMaterial
        color="#94a3b8"
        transparent
        opacity={0.15}
        wireframe
      />
    </mesh>
  ) : null;
}

function CameraController({ bounds }: { bounds: ToolpathPreviewProps["modelBounds"] }) {
  const { camera } = useThree();

  useEffect(() => {
    if (bounds) {
      const { minX, maxX, minY, maxY, minZ, maxZ } = bounds;
      const centerX = (minX + maxX) / 2;
      const centerY = (minY + maxY) / 2;
      const centerZ = (minZ + maxZ) / 2;

      const size = Math.max(maxX - minX, maxY - minY, maxZ - minZ) * 1.5;

      camera.position.set(centerX + size, centerZ + size * 0.8, centerY + size);
      camera.lookAt(centerX, centerZ, centerY);
    }
  }, [bounds, camera]);

  return null;
}

// 解析G代码，提取刀具路径
function parseGcode(gcode: string): { positions: THREE.Vector3[]; rapid: boolean[] } {
  const positions: THREE.Vector3[] = [];
  const rapid: boolean[] = [];
  const lines = gcode.split('\n');
  
  let currentPos = new THREE.Vector3(0, 0, 0);
  
  for (const line of lines) {
    const trimmed = line.trim();
    if (!trimmed || trimmed.startsWith(';')) continue;
    
    const isRapid = trimmed.includes('G0') || trimmed.includes('G00');
    const xMatch = trimmed.match(/X([-+]?[0-9]*\.?[0-9]+)/);
    const yMatch = trimmed.match(/Y([-+]?[0-9]*\.?[0-9]+)/);
    const zMatch = trimmed.match(/Z([-+]?[0-9]*\.?[0-9]+)/);
    
    const newPos = currentPos.clone();
    
    if (xMatch) newPos.x = parseFloat(xMatch[1]);
    if (yMatch) newPos.y = parseFloat(yMatch[1]);
    if (zMatch) newPos.z = parseFloat(zMatch[1]);
    
    // 避免重复点
    if (newPos.distanceTo(currentPos) > 0.001) {
      positions.push(currentPos.clone());
      rapid.push(isRapid);
      currentPos = newPos;
    }
  }
  
  // 添加最后一点
  if (positions.length > 0) {
    positions.push(currentPos.clone());
  }
  
  return { positions, rapid };
}

// 刀具路径线段组件
function Toolpath({ positions, rapid, lineColor, rapidColor }: {
  positions: THREE.Vector3[];
  rapid: boolean[];
  lineColor: string;
  rapidColor: string;
}) {
  const lineRef = useRef<THREE.LineSegments>(null);
  const [animationProgress, setAnimationProgress] = useState(0);
  const [isAnimating, setIsAnimating] = useState(false);
  
  const { geometry, colors } = useMemo(() => {
    const points: THREE.Vector3[] = [];
    const lineColors: THREE.Color[] = [];
    
    const color1 = new THREE.Color(lineColor);
    const color2 = new THREE.Color(rapidColor);
    
    for (let i = 0; i < positions.length - 1; i++) {
      points.push(positions[i]);
      points.push(positions[i + 1]);
      
      const color = rapid[i] ? color2 : color1;
      lineColors.push(color, color);
    }
    
    const geo = new THREE.BufferGeometry().setFromPoints(points);
    geo.setAttribute('color', new THREE.Float32BufferAttribute([
      ...lineColors.flatMap(c => [c.r, c.g, c.b])
    ], 3));
    
    return { geometry: geo, colors: lineColors };
  }, [positions, rapid, lineColor, rapidColor]);
  
  useEffect(() => {
    setAnimationProgress(0);
    setIsAnimating(true);
  }, [positions]);
  
  useFrame((_, delta) => {
    if (isAnimating && animationProgress < 1) {
      setAnimationProgress(prev => Math.min(prev + delta * 0.5, 1));
    }
  });
  
  const visibleCount = Math.floor(positions.length * animationProgress);
  
  return (
    <lineSegments ref={lineRef} geometry={geometry}>
      <lineBasicMaterial vertexColors transparent opacity={0.9} />
    </lineSegments>
  );
}

// 场景控制器
function SceneController({ positions }: { positions: THREE.Vector3[] }) {
  const { camera } = useThree();
  
  useEffect(() => {
    if (positions.length === 0) return;
    
    // 计算边界框
    const box = new THREE.Box3();
    positions.forEach(p => box.expandByPoint(p));
    
    const center = box.getCenter(new THREE.Vector3());
    const size = box.getSize(new THREE.Vector3());
    const maxDim = Math.max(size.x, size.y, size.z);
    
    // 设置相机位置
    const distance = maxDim * 2.5;
    camera.position.set(center.x + distance, center.y + distance * 0.8, center.z + distance);
    camera.lookAt(center);
  }, [positions, camera]);
  
  return null;
}

// 路径统计
function PathStats({ positions, rapid }: { positions: THREE.Vector3[]; rapid: boolean[] }) {
  const stats = useMemo(() => {
    if (positions.length < 2) return null;
    
    let totalLength = 0;
    let rapidLength = 0;
    let cutLength = 0;
    
    for (let i = 0; i < positions.length - 1; i++) {
      const segmentLength = positions[i].distanceTo(positions[i + 1]);
      totalLength += segmentLength;
      if (rapid[i]) {
        rapidLength += segmentLength;
      } else {
        cutLength += segmentLength;
      }
    }
    
    return {
      totalLength: totalLength.toFixed(2),
      rapidLength: rapidLength.toFixed(2),
      cutLength: cutLength.toFixed(2),
      lineCount: positions.length - 1,
      rapidCount: rapid.filter(Boolean).length,
      cutCount: rapid.filter(r => !r).length,
    };
  }, [positions, rapid]);
  
  if (!stats) return null;
  
  return (
    <div className="absolute bottom-4 left-4 bg-background/95 backdrop-blur-sm border border-border rounded-lg p-4 shadow-lg">
      <h3 className="text-sm font-semibold mb-3 text-foreground">刀路统计</h3>
      <div className="grid grid-cols-2 gap-x-6 gap-y-2 text-xs">
        <span className="text-muted-foreground">总长度</span>
        <span className="font-mono text-foreground">{stats.totalLength} mm</span>
        <span className="text-muted-foreground">切削长度</span>
        <span className="font-mono text-green-600">{stats.cutLength} mm</span>
        <span className="text-muted-foreground">快进长度</span>
        <span className="font-mono text-amber-600">{stats.rapidLength} mm</span>
        <span className="text-muted-foreground">线段总数</span>
        <span className="font-mono text-foreground">{stats.lineCount}</span>
        <span className="text-muted-foreground">切削次数</span>
        <span className="font-mono text-green-600">{stats.cutCount}</span>
        <span className="text-muted-foreground">快进次数</span>
        <span className="font-mono text-amber-600">{stats.rapidCount}</span>
      </div>
    </div>
  );
}
